\section{Objective}
Build a distributed IoT system using two ESP32 LoRa nodes that detect traffic on two sides of an intersection using ultrasonic sensors and dynamically control traffic lights based on real-time traffic conditions.

\section{Hardware Setup}

\subsection{Microcontrollers}
\begin{itemize}
    \item 2 $\times$ Heltec WiFi LoRa 32 V3 (ESP32-S3)
    \item Node A: Side A sensing + LEDs
    \item Node B: Side B sensing + controller (or shared control)
\end{itemize}

\subsection{Sensors}
\begin{itemize}
    \item 4 $\times$ Ultrasonic sensors (HC-SR04)
    \item Per side:
    \begin{itemize}
        \item 1 Far sensor (incoming cars)
        \item 1 Near sensor (queue detection)
    \end{itemize}
\end{itemize}

\subsection{Actuators}
\begin{itemize}
    \item 6 $\times$ LEDs (traffic lights)
    \begin{itemize}
        \item 2 Red
        \item 2 Yellow
        \item 2 Green
    \end{itemize}
\end{itemize}

\subsection{Other Components}
\begin{itemize}
    \item Breadboard
    \item Jumper wires
    \item Resistors
    \item Power supply (battery)
\end{itemize}

\section{System Architecture}

\begin{itemize}
    \item Each node reads its two ultrasonic sensors
    \item Each node estimates:
    \begin{itemize}
        \item Incoming vehicles (far sensor)
        \item Queue presence (near sensor)
    \end{itemize}
    \item Nodes communicate using LoRa
    \item One node acts as controller:
    \begin{itemize}
        \item Receives data from both sides
        \item Computes traffic decision
        \item Controls traffic lights
    \end{itemize}
\end{itemize}

\section{Data Processing}

Each node computes:
\begin{itemize}
    \item Incoming count
    \item Queue presence
    \item Queue estimate:
\end{itemize}

\[
\text{Queue} \approx \text{Incoming} - \text{Passed}
\]

\section{Decision Algorithm}

\begin{lstlisting}
IF current side has cars waiting:
    keep green
ELSE IF other side has cars:
    switch green
ELSE:
    alternate normally
\end{lstlisting}

Optional improvement:
\begin{lstlisting}
extend green if queue > threshold
\end{lstlisting}

\section{Real-Time Behavior}

\begin{itemize}
    \item Sensors read continuously
    \item Data updated every $\sim$1 second
    \item LoRa transmission:
    \begin{itemize}
        \item periodic OR
        \item event-driven
    \end{itemize}
\end{itemize}

\section{Communication (LoRa)}

\begin{itemize}
    \item Low bandwidth communication
    \item Small packets exchanged
\end{itemize}

Example message:
\begin{lstlisting}
{
  side: "A",
  incoming: 5,
  queue: 3
}
\end{lstlisting}

\section{Traffic Light Control}

Each side follows:
\begin{center}
GREEN $\rightarrow$ YELLOW $\rightarrow$ RED
\end{center}

Controller adjusts:
\begin{itemize}
    \item Green duration dynamically
    \item Immediate switch if no cars detected
\end{itemize}

\section{Constraints}

\begin{itemize}
    \item Frequency: Continuous updates during cycle
    \item Bandwidth: Very low (small packets)
    \item Latency: 1--2 seconds
    \item Energy: $\sim$60--80 mA per node (battery-powered)
    \item Duty Cycle: Controlled transmissions
\end{itemize}

\section{Expected Behavior}

\begin{itemize}
    \item If one side is empty $\rightarrow$ keep green on the other side
    \item If both sides have traffic $\rightarrow$ compare queues
    \item If queue cleared $\rightarrow$ switch immediately
    \item Fully adaptive real-time system
\end{itemize}
